    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const editorPopup = document.getElementById('editor-popup');
            const editorTextarea = document.getElementById('editor-textarea');
            const saveBtn = document.getElementById('save-btn');
            const cancelBtn = document.getElementById('cancel-btn');
            let currentElement = null;

            document.querySelectorAll('.editable-content').forEach(element => {
                element.addEventListener('click', (e) => {
                    e.stopPropagation();
                    currentElement = e.target;
                    editorTextarea.value = currentElement.textContent.trim();
                    editorPopup.classList.remove('hidden');
                });
            });

            saveBtn.addEventListener('click', () => {
                if (currentElement) {
                    currentElement.textContent = editorTextarea.value;
                }
                editorPopup.classList.add('hidden');
            });

            cancelBtn.addEventListener('click', () => {
                editorPopup.classList.add('hidden');
            });

            document.addEventListener('click', (e) => {
                if (!editorPopup.contains(e.target) && !e.target.classList.contains('editable-content')) {
                    editorPopup.classList.add('hidden');
                }
            });
        });
    </script>
